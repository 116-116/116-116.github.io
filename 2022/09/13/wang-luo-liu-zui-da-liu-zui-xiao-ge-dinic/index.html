网络流最大流-最小割-dinic

#include<bits/stdc++.h>
#define ll  long long
using namespace std;
const int N = 210,M=5100;
ll head[N], ed[M * 2], nex[M * 2], idx = 1, val[M * 2]; // idx=1 使编号从2，3开始  正反边通过^1可得
ll dep[N],cur[N];//dep存深度（层级），防止dfs搜的太深，cur存u点的当前出边
ll n,s,t,m;
ll a, b, c;
int vis[N];//记录最小割的集合
void add(ll a,ll b,ll c)
{
    ed[++idx] = b;
    val[idx] = c;
    nex[idx] = head[a];
    head[a] = idx;
}
bool bfs()
{
    memset(dep, 0, (n + 2) * sizeof dep[0]);//找增广路并重新分层
    queue<int> q;
    q.push(s);
    dep[s] = 1;
    while(q.size())
    {
        int u = q.front();
        q.pop();
        for (int i = head[u]; i;i=nex[i])
        {
            int v = ed[i];
            if(!dep[v]&&val[i])
            {
                dep[v] = dep[u] + 1;
                q.push(v);
                if(v==t)
                    return 1;
            }
        }
    }
    return 0;
}
ll dfs(int u,ll mf)//多路增广
{
    if(u==t)
        return mf;
    ll sum = 0;
    for (int i = cur[u]; i;i=nex[i])
    {
        cur[u] = i;//当前弧优化，之前的边已无法流通，不再遍历
        int v = ed[i];
        if(dep[v]==dep[u]+1&&val[i])
        {
            ll f = dfs(v, min(mf, val[i]));
            val[i] -= f;
            val[i ^ 1] += f;//更新残留网
            sum += f;//累加u点的流出流量
            mf -= f;//更新u点的剩余流量
            if(!mf)//余量优化：u点流量耗尽，结束对u的增广
                break;
        }
    }
    if(!sum)//残枝优化，sum=0 此前路径无法流通可以删除
        dep[u] = 0;
    return sum;
}
ll dinic()//累加可以流通的流量
{
    ll flow = 0;
    while(bfs())
    {
        memcpy(cur, head, (n + 5) * sizeof head[0]);
        flow += dfs(s, 1e18);
    }
    return flow;
}
void mincut(int u)
{
    vis[u] = 1;
    for (int i = head[u]; i;i=nex[i])
    {
        if(!vis[ed[i]]&&val[i])
            mincut(ed[i]);
    }
}
int main()
{
    scanf("%lld%lld%lld%lld", &n, &m, &s, &t);
    for (int i = 1; i <= m;++i)
    {
        scanf("%lld%lld%lld", &a, &b, &c);
        add(a, b, c);
        add(b, a, 0);
    }
    cout << dinic() << endl;//最大流-最小割

    // for (int i = 1; i <= m;++i)
    // {
    //     scanf("%lld%lld", &a, &b);
    //     add(a, b, 1);
    //     add(b, a, 0);
    // }
    // cout << dinic() << endl;//最小割的最少边数

    mincut(s);
    vector<int> s_set, t_set;//最小割的划分之一
    for (int i = 1; i <= n;++i)
        if(vis[i])
            s_set.push_back(i);
        else
            t_set.push_back(i);
        // system("pause");
        return 0;
}
